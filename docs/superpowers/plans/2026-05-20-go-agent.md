# Go Agent 实现计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 用 Go 实现跨平台的 SuperDev agent，支持读取配置文件、管理服务进程、收集日志写入 SQLite、并通过 HTTP REST + WebSocket 暴露 API。

**Architecture:** 单二进制 Go 程序，分为 config（配置读写）、model（数据结构）、process（进程管理）、store（SQLite 日志存储）、api（HTTP/WebSocket 服务）五个包。进程捕获 stdout/stderr 逐行送入内存缓冲，批量写入 SQLite；WebSocket 客户端从内存缓冲实时推送。

**Tech Stack:** Go 1.22+、`gopkg.in/yaml.v3`（YAML 配置）、`modernc.org/sqlite`（纯 Go SQLite，无 CGO）、`github.com/gorilla/websocket`（WebSocket）、`github.com/google/uuid`（UUID 生成）、`github.com/stretchr/testify`（测试断言）

`agent/config/loader.go`:
```go
package config

import (
	"errors"
	"fmt"
	"os"
	"path/filepath"

	"github.com/google/uuid"
	"github.com/superdev/agent/model"
	"gopkg.in/yaml.v3"
)

var ErrNotFound = errors.New("config file not found")

type Loader struct {
	rootPath string
}

func NewLoader(rootPath string) *Loader {
	return &Loader{rootPath: rootPath}
}

func (l *Loader) configPath() string {
	return filepath.Join(l.rootPath, ".superdev", "config.yaml")
}

func (l *Loader) Load() (model.Project, error) {
	data, err := os.ReadFile(l.configPath())
	if errors.Is(err, os.ErrNotExist) {
		return model.Project{}, ErrNotFound
	}
	if err != nil {
		return model.Project{}, fmt.Errorf("read config: %w", err)
	}

	var raw struct {
		Name               string          `yaml:"name"`
		Services           []serviceYAML   `yaml:"services"`
		SelectedServiceIDs []string        `yaml:"selected_service_ids"`
	}
	if err := yaml.Unmarshal(data, &raw); err != nil {
		return model.Project{}, fmt.Errorf("parse config: %w", err)
	}

	services := make([]model.Service, len(raw.Services))
	for i, s := range raw.Services {
		services[i] = model.Service{
			ID:       uuid.NewString(),
			Name:     s.Name,
			Command:  s.Command,
			WorkDir:  s.WorkingDir,
			Required: s.Required,
			Order:    s.Order,
			EnvFile:  s.EnvFile,
			Env:      s.Env,
			Status:   model.StatusStopped,
		}
	}

	return model.Project{
		ID:                 uuid.NewString(),
		Name:               raw.Name,
		RootPath:           l.rootPath,
		Services:           services,
		SelectedServiceIDs: raw.SelectedServiceIDs,
	}, nil
}

func (l *Loader) Save(p model.Project) error {
	dir := filepath.Dir(l.configPath())
	if err := os.MkdirAll(dir, 0o755); err != nil {
		return err
	}

	// 保留现有 log_rules，避免 Save 时丢失
	existing := map[string]interface{}{}
	if data, err := os.ReadFile(l.configPath()); err == nil {
		_ = yaml.Unmarshal(data, &existing)
	}

	raw := map[string]interface{}{
		"name":                 p.Name,
		"services":             servicesToYAML(p.Services),
		"selected_service_ids": p.SelectedServiceIDs,
	}
	if lr, ok := existing["log_rules"]; ok {
		raw["log_rules"] = lr
	}

	data, err := yaml.Marshal(raw)
	if err != nil {
		return fmt.Errorf("marshal config: %w", err)
	}
	return os.WriteFile(l.configPath(), data, 0o644)
}

func (l *Loader) LoadLogRules() ([]model.LogRule, error) {
	data, err := os.ReadFile(l.configPath())
	if errors.Is(err, os.ErrNotExist) {
		return nil, nil
	}
	if err != nil {
		return nil, err
	}
	var raw struct {
		LogRules []model.LogRule `yaml:"log_rules"`
	}
	if err := yaml.Unmarshal(data, &raw); err != nil {
		return nil, err
	}
	return raw.LogRules, nil
}

func (l *Loader) SaveLogRules(rules []model.LogRule) error {
	existing := map[string]interface{}{}
	if data, err := os.ReadFile(l.configPath()); err == nil {
		_ = yaml.Unmarshal(data, &existing)
	}
	existing["log_rules"] = rules
	data, err := yaml.Marshal(existing)
	if err != nil {
		return err
	}
	dir := filepath.Dir(l.configPath())
	if err := os.MkdirAll(dir, 0o755); err != nil {
		return err
	}
	return os.WriteFile(l.configPath(), data, 0o644)
}

// serviceYAML 对应 YAML 文件中的 snake_case 字段
type serviceYAML struct {
	Name       string            `yaml:"name"`
	Command    string            `yaml:"command"`
	WorkingDir string            `yaml:"working_dir"`
	Required   bool              `yaml:"required"`
	Order      int               `yaml:"order"`
	EnvFile    string            `yaml:"env_file,omitempty"`
	Env        map[string]string `yaml:"env,omitempty"`
}

func servicesToYAML(services []model.Service) []serviceYAML {
	out := make([]serviceYAML, len(services))
	for i, s := range services {
		out[i] = serviceYAML{
			Name:       s.Name,
			Command:    s.Command,
			WorkingDir: s.WorkDir,
			Required:   s.Required,
			Order:      s.Order,
			EnvFile:    s.EnvFile,
			Env:        s.Env,
		}
	}
	return out
}
```

package config

import (
	"encoding/json"
	"os"
	"path/filepath"
	"slices"
	"sync"
)

type Registry struct {
	mu   sync.Mutex
	path string
}

func NewRegistry(path string) *Registry {
	return &Registry{path: path}
}

func (r *Registry) List() []string {
	r.mu.Lock()
	defer r.mu.Unlock()
	paths, _ := r.load()
	return paths
}

func (r *Registry) Add(rootPath string) error {
	r.mu.Lock()
	defer r.mu.Unlock()
	paths, _ := r.load()
	if !slices.Contains(paths, rootPath) {
		paths = append(paths, rootPath)
	}
	return r.save(paths)
}

func (r *Registry) Remove(rootPath string) error {
	r.mu.Lock()
	defer r.mu.Unlock()
	paths, _ := r.load()
	paths = slices.DeleteFunc(paths, func(p string) bool { return p == rootPath })
	return r.save(paths)
}

func (r *Registry) load() ([]string, error) {
	data, err := os.ReadFile(r.path)
	if os.IsNotExist(err) {
		return nil, nil
	}
	if err != nil {
		return nil, err
	}
	var paths []string
	if err := json.Unmarshal(data, &paths); err != nil {
		return nil, err
	}
	return paths, nil
}

func (r *Registry) save(paths []string) error {
	if err := os.MkdirAll(filepath.Dir(r.path), 0o755); err != nil {
		return err
	}
	data, err := json.Marshal(paths)
	if err != nil {
		return err
	}
	return os.WriteFile(r.path, data, 0o644)
}
```

func TestFetchPagination(t *testing.T) {
	s := newTestStore(t)
	now := time.Now().UTC()
	entries := make([]model.LogEntry, 5)
	for i := range entries {
		entries[i] = model.LogEntry{
			ServiceID: "svc-1", RunID: "run-1",
			Timestamp: now.Add(time.Duration(i) * time.Second),
			Level: "INFO", Message: "msg", Stream: "stdout",
		}

	require.NoError(t, s.AppendBatch(entries))

	first, err := s.Fetch(store.FetchParams{ServiceID: "svc-1", Limit: 3})
	require.NoError(t, err)
	assert.Len(t, first, 3)

	// before cursor: get entries with ID < first[0].ID (older)
	second, err := s.Fetch(store.FetchParams{ServiceID: "svc-1", Limit: 3, Before: first[0].ID})

package store

import (
	"database/sql"
	"fmt"
	"time"

	"github.com/superdev/agent/model"
	_ "modernc.org/sqlite"
)

type Store struct {
	db *sql.DB
}

type FetchParams struct {
	ServiceID string
	RunID     string
	Limit     int
	Before    int64 // 返回 ID < Before 的条目（旧数据分页），0 表示不限制
}

func New(path string) (*Store, error) {
	db, err := sql.Open("sqlite", path)
	if err != nil {
		return nil, fmt.Errorf("open sqlite: %w", err)
	}
	db.SetMaxOpenConns(1) // SQLite 不支持并发写，单连接序列化
	if err := migrate(db); err != nil {
		return nil, fmt.Errorf("migrate: %w", err)
	}
	return &Store{db: db}, nil
}

func (s *Store) Close() error { return s.db.Close() }

func migrate(db *sql.DB) error {
	_, err := db.Exec(`
		CREATE TABLE IF NOT EXISTS log_entries (
			id         INTEGER PRIMARY KEY AUTOINCREMENT,
			service_id TEXT    NOT NULL,
			run_id     TEXT    NOT NULL,
			timestamp  DATETIME NOT NULL,
			level      TEXT    NOT NULL,
			message    TEXT    NOT NULL,
			stream     TEXT    NOT NULL
		);
		CREATE INDEX IF NOT EXISTS idx_service_id ON log_entries(service_id);
		CREATE INDEX IF NOT EXISTS idx_run_id     ON log_entries(run_id);
		CREATE INDEX IF NOT EXISTS idx_timestamp  ON log_entries(timestamp);
	`)
	return err
}

func (s *Store) AppendBatch(entries []model.LogEntry) error {
	if len(entries) == 0 {
		return nil
	}
	tx, err := s.db.Begin()
	if err != nil {
		return err
	}
	stmt, err := tx.Prepare(`
		INSERT INTO log_entries (service_id, run_id, timestamp, level, message, stream)
		VALUES (?, ?, ?, ?, ?, ?)
	`)
	if err != nil {
		_ = tx.Rollback()
		return err
	}
	defer stmt.Close()

	for _, e := range entries {
		if _, err := stmt.Exec(e.ServiceID, e.RunID, e.Timestamp.UTC(), e.Level, e.Message, e.Stream); err != nil {
			_ = tx.Rollback()
			return err
		}
	}
	return tx.Commit()
}

func (s *Store) Fetch(p FetchParams) ([]model.LogEntry, error) {
	if p.Limit <= 0 {
		p.Limit = 1000
	}
	query := `SELECT id, service_id, run_id, timestamp, level, message, stream FROM log_entries WHERE 1=1`
	args := []any{}

	if p.ServiceID != "" {
		query += " AND service_id = ?"
		args = append(args, p.ServiceID)
	}
	if p.RunID != "" {
		query += " AND run_id = ?"
		args = append(args, p.RunID)
	}
	if p.Before > 0 {
		query += " AND id < ?"
		args = append(args, p.Before)
	}
	query += fmt.Sprintf(" ORDER BY id ASC LIMIT %d", p.Limit)

	rows, err := s.db.Query(query, args...)
	if err != nil {
		return nil, err
	}
	defer rows.Close()

	var entries []model.LogEntry
	for rows.Next() {
		var e model.LogEntry
		var ts string
		if err := rows.Scan(&e.ID, &e.ServiceID, &e.RunID, &ts, &e.Level, &e.Message, &e.Stream); err != nil {
			return nil, err
		}
		e.Timestamp, _ = time.Parse("2006-01-02T15:04:05.999999999Z07:00", ts)
		if e.Timestamp.IsZero() {
			e.Timestamp, _ = time.Parse("2006-01-02 15:04:05.999999999+00:00", ts)
		}
		entries = append(entries, e)
	}
	return entries, rows.Err()
}

func (s *Store) DeleteOlderThan(days int) error {
	cutoff := time.Now().UTC().AddDate(0, 0, -days)
	_, err := s.db.Exec("DELETE FROM log_entries WHERE timestamp < ?", cutoff)
	return err
}

package logbuf

import (
	"sync"
	"time"

	"github.com/superdev/agent/model"
	"github.com/superdev/agent/store"
)

const (
	flushInterval = 100 * time.Millisecond
	flushBatch    = 50
)

// Flusher 是 store.Store 的最小接口，便于测试时注入 nil。
type Flusher interface {
	AppendBatch([]model.LogEntry) error
}

type Buffer struct {
	mu          sync.RWMutex
	ring        []model.LogEntry // 环形缓冲
	head        int              // 下一个写入位置
	count       int              // 已写入总数（含溢出）
	maxSize     int
	pending     []model.LogEntry // 待 flush 到 store 的批次
	subs        map[string]chan model.LogEntry
	store       Flusher
	done        chan struct{}
}

func New(store Flusher, maxSize int) *Buffer {
	b := &Buffer{
		ring:    make([]model.LogEntry, maxSize),
		maxSize: maxSize,
		subs:    map[string]chan model.LogEntry{},
		store:   store,
		done:    make(chan struct{}),
	}
	go b.flushLoop()
	return b
}

func (b *Buffer) Append(e model.LogEntry) {
	b.mu.Lock()
	b.ring[b.head] = e
	b.head = (b.head + 1) % b.maxSize
	b.count++
	b.pending = append(b.pending, e)
	shouldFlush := len(b.pending) >= flushBatch
	subs := make([]chan model.LogEntry, 0, len(b.subs))
	for _, ch := range b.subs {
		subs = append(subs, ch)
	}
	b.mu.Unlock()

	// 通知订阅者（非阻塞）
	for _, ch := range subs {
		select {
		case ch <- e:
		default:
		}
	}

	if shouldFlush {
		b.flush()
	}
}

// Recent 返回最近 n 条日志（按时间正序）。
func (b *Buffer) Recent(n int) []model.LogEntry {
	b.mu.RLock()
	defer b.mu.RUnlock()
	total := b.count
	if total > b.maxSize {
		total = b.maxSize
	}
	if n > total {
		n = total
	}
	out := make([]model.LogEntry, n)
	start := (b.head - n + b.maxSize) % b.maxSize
	for i := 0; i < n; i++ {
		out[i] = b.ring[(start+i)%b.maxSize]
	}
	return out
}

func (b *Buffer) Subscribe(id string) <-chan model.LogEntry {
	b.mu.Lock()
	defer b.mu.Unlock()
	ch := make(chan model.LogEntry, 64)
	b.subs[id] = ch
	return ch
}

func (b *Buffer) Unsubscribe(id string) {
	b.mu.Lock()
	defer b.mu.Unlock()
	if ch, ok := b.subs[id]; ok {
		close(ch)
		delete(b.subs, id)
	}
}

func (b *Buffer) Close() {
	close(b.done)
	b.flush()
}

func (b *Buffer) flushLoop() {
	ticker := time.NewTicker(flushInterval)
	defer ticker.Stop()
	for {
		select {
		case <-ticker.C:
			b.flush()
		case <-b.done:
			return
		}
	}
}

func (b *Buffer) flush() {
	b.mu.Lock()
	if len(b.pending) == 0 {
		b.mu.Unlock()
		return
	}
	batch := b.pending
	b.pending = nil
	b.mu.Unlock()

	if b.store != nil {
		_ = b.store.AppendBatch(batch)
	}
}
```

package process

import (
	"bufio"
	"os"
	"os/exec"
	"sync"
)

type RunnerConfig struct {
	Command string
	WorkDir string
	Env     map[string]string
	EnvFile string
	OnLine  func(line, stream string)
}

type Runner struct {
	cfg  RunnerConfig
	mu   sync.Mutex
	cmd  *exec.Cmd
}

func NewRunner(cfg RunnerConfig) *Runner {
	return &Runner{cfg: cfg}
}

func (r *Runner) Start() error {
	r.mu.Lock()
	defer r.mu.Unlock()

	cmd := exec.Command("sh", "-c", r.cfg.Command)
	cmd.Dir = r.cfg.WorkDir
	cmd.Env = r.buildEnv()

	stdout, err := cmd.StdoutPipe()
	if err != nil {
		return err
	}
	stderr, err := cmd.StderrPipe()
	if err != nil {
		return err
	}

	if err := cmd.Start(); err != nil {
		return err
	}
	r.cmd = cmd

	go r.scanLines(bufio.NewScanner(stdout), "stdout")
	go r.scanLines(bufio.NewScanner(stderr), "stderr")
	go func() { _ = cmd.Wait() }()

	return nil
}

func (r *Runner) Stop() {
	r.mu.Lock()
	cmd := r.cmd
	r.mu.Unlock()
	if cmd != nil && cmd.Process != nil {
		_ = cmd.Process.Kill()
	}
}

func (r *Runner) IsRunning() bool {
	r.mu.Lock()
	defer r.mu.Unlock()
	return r.cmd != nil && r.cmd.ProcessState == nil
}

func (r *Runner) PID() int {
	r.mu.Lock()
	defer r.mu.Unlock()
	if r.cmd != nil && r.cmd.Process != nil {
		return r.cmd.Process.Pid
	}
	return 0
}

func (r *Runner) scanLines(scanner *bufio.Scanner, stream string) {
	for scanner.Scan() {
		r.cfg.OnLine(scanner.Text(), stream)
	}
}

func (r *Runner) buildEnv() []string {
	base := os.Environ()
	for k, v := range r.cfg.Env {
		base = append(base, k+"="+v)
	}
	return base
}

package process

import (
	"slices"
	"sync"
	"time"

	"github.com/superdev/agent/model"
)

type Manager struct {
	mu      sync.Mutex
	runners map[string]*Runner
	status  map[string]model.ServiceStatus
	onLog   func(model.LogEntry)
	runID   string
}

func NewManager(onLog func(model.LogEntry)) *Manager {
	return &Manager{
		runners: map[string]*Runner{},
		status:  map[string]model.ServiceStatus{},
		onLog:   onLog,
	}
}

func (m *Manager) SetRunID(id string) {
	m.mu.Lock()
	m.runID = id
	m.mu.Unlock()
}

func (m *Manager) Start(svc model.Service) error {
	m.setStatus(svc.ID, model.StatusStarting)
	r := NewRunner(RunnerConfig{
		Command: svc.Command,
		WorkDir: svc.WorkDir,
		Env:     svc.Env,
		EnvFile: svc.EnvFile,
		OnLine: func(line, stream string) {
			m.mu.Lock()
			runID := m.runID
			m.mu.Unlock()
			m.onLog(model.LogEntry{
				ServiceID: svc.ID,
				RunID:     runID,
				Timestamp: time.Now().UTC(),
				Level:     "INFO",
				Message:   line,
				Stream:    stream,
			})
		},
	})

	if err := r.Start(); err != nil {
		m.setStatus(svc.ID, model.StatusFailed)
		return err
	}

	m.mu.Lock()
	m.runners[svc.ID] = r
	m.mu.Unlock()
	m.setStatus(svc.ID, model.StatusRunning)

	// 监控进程退出
	go func() {
		for r.IsRunning() {
			time.Sleep(200 * time.Millisecond)
		}
		m.setStatus(svc.ID, model.StatusStopped)
	}()

	return nil
}

// StartGroup 按 order 分组串行启动，同组并行。
func (m *Manager) StartGroup(services []model.Service) error {
	groups := groupByOrder(services)
	for _, group := range groups {
		var wg sync.WaitGroup
		errCh := make(chan error, len(group))
		for _, svc := range group {
			wg.Add(1)
			go func(s model.Service) {
				defer wg.Done()
				if err := m.Start(s); err != nil {
					errCh <- err
				}
			}(svc)
		}
		wg.Wait()
		close(errCh)
		for err := range errCh {
			return err // 返回第一个错误，停止后续 group
		}
	}
	return nil
}

func (m *Manager) Stop(serviceID string) {
	m.mu.Lock()
	r := m.runners[serviceID]
	m.mu.Unlock()
	if r != nil {
		r.Stop()
	}
	m.setStatus(serviceID, model.StatusStopped)
}

func (m *Manager) StopAll() {
	m.mu.Lock()
	ids := make([]string, 0, len(m.runners))
	for id := range m.runners {
		ids = append(ids, id)
	}
	m.mu.Unlock()
	for _, id := range ids {
		m.Stop(id)
	}
}

func (m *Manager) Status(serviceID string) model.ServiceStatus {
	m.mu.Lock()
	defer m.mu.Unlock()
	if st, ok := m.status[serviceID]; ok {
		return st
	}
	return model.StatusStopped
}

func (m *Manager) PID(serviceID string) int {
	m.mu.Lock()
	r := m.runners[serviceID]
	m.mu.Unlock()
	if r != nil {
		return r.PID()
	}
	return 0
}

func (m *Manager) setStatus(id string, st model.ServiceStatus) {
	m.mu.Lock()
	m.status[id] = st
	m.mu.Unlock()
}

// groupByOrder 按 order 升序分组，返回有序的分组列表。
func groupByOrder(services []model.Service) [][]model.Service {
	orders := []int{}
	byOrder := map[int][]model.Service{}
	for _, s := range services {
		if _, ok := byOrder[s.Order]; !ok {
			orders = append(orders, s.Order)
		}
		byOrder[s.Order] = append(byOrder[s.Order], s)
	}
	slices.Sort(orders)
	groups := make([][]model.Service, len(orders))
	for i, o := range orders {
		groups[i] = byOrder[o]
	}
	return groups
}

package api

import (
	"net/http"
	"path/filepath"
	"sync"

	"github.com/superdev/agent/config"
	"github.com/superdev/agent/logbuf"
	"github.com/superdev/agent/model"
	"github.com/superdev/agent/process"
	"github.com/superdev/agent/store"
)

type AppConfig struct {
	DataDir string // agent 数据目录，存放 projects.json 和 logs.db
}

type App struct {
	cfg      AppConfig
	mu       sync.RWMutex
	projects []model.Project           // 运行时项目列表（含服务运行时状态）
	managers map[string]*process.Manager // projectID → manager
	buf      *logbuf.Buffer
	store    *store.Store
	registry *config.Registry
}

func NewApp(cfg AppConfig) *App {
	st, _ := store.New(filepath.Join(cfg.DataDir, "logs.db"))
	buf := logbuf.New(st, 8000)
	return &App{
		cfg:      cfg,
		managers: map[string]*process.Manager{},
		buf:      buf,
		store:    st,
		registry: config.NewRegistry(filepath.Join(cfg.DataDir, "projects.json")),
	}
}

func (a *App) Handler() http.Handler {
	mux := http.NewServeMux()
	mux.HandleFunc("GET /api/projects", a.listProjects)
	mux.HandleFunc("POST /api/projects", a.addProject)
	mux.HandleFunc("DELETE /api/projects/{id}", a.removeProject)
	mux.HandleFunc("GET /api/projects/{id}/rules", a.getProjectRules)
	mux.HandleFunc("PUT /api/projects/{id}/rules", a.putProjectRules)
	mux.HandleFunc("GET /api/services", a.listServices)
	mux.HandleFunc("POST /api/services/{id}/start", a.startService)
	mux.HandleFunc("POST /api/services/{id}/stop", a.stopService)
	mux.HandleFunc("POST /api/services/{id}/restart", a.restartService)
	mux.HandleFunc("POST /api/projects/{id}/start-selected", a.startSelected)
	mux.HandleFunc("GET /api/logs", a.fetchLogs)
	mux.HandleFunc("GET /ws/logs", a.wsLogs)
	return mux
}

func (a *App) Start(addr string) error {
	a.loadRegisteredProjects()
	return http.ListenAndServe(addr, a.Handler())
}

func (a *App) loadRegisteredProjects() {
	paths := a.registry.List()
	for _, path := range paths {
		loader := config.NewLoader(path)
		p, err := loader.Load()
		if err != nil {
			continue
		}
		a.mu.Lock()
		a.projects = append(a.projects, p)
		a.mu.Unlock()
	}
}

package api

import (
	"encoding/json"

func (a *App) listProjects(w http.ResponseWriter, r *http.Request) {
	a.mu.RLock()
	projects := make([]model.Project, len(a.projects))
	copy(projects, a.projects)
	a.mu.RUnlock()
	jsonOK(w, projects)
}

func (a *App) addProject(w http.ResponseWriter, r *http.Request) {
	var req struct {
		RootPath string `json:"root_path"`
	}
	if err := json.NewDecoder(r.Body).Decode(&req); err != nil || req.RootPath == "" {
		http.Error(w, "invalid request", http.StatusBadRequest)
		return
	}

	loader := config.NewLoader(req.RootPath)
	p, err := loader.Load()
	if err != nil {
		http.Error(w, "failed to load config: "+err.Error(), http.StatusBadRequest)
		return
	}

	if err := a.registry.Add(req.RootPath); err != nil {
		http.Error(w, "registry error", http.StatusInternalServerError)
		return
	}

	a.mu.Lock()
	a.projects = append(a.projects, p)
	a.mu.Unlock()

	jsonOK(w, p)
}

func (a *App) removeProject(w http.ResponseWriter, r *http.Request) {
	id := r.PathValue("id")
	a.mu.Lock()
	var rootPath string
	newProjects := a.projects[:0]
	for _, p := range a.projects {
		if p.ID == id {
			rootPath = p.RootPath
		} else {
			newProjects = append(newProjects, p)
		}
	}
	a.projects = newProjects
	a.mu.Unlock()

	if rootPath != "" {
		_ = a.registry.Remove(rootPath)
	}
	w.WriteHeader(http.StatusNoContent)
}

func (a *App) getProjectRules(w http.ResponseWriter, r *http.Request) {
	id := r.PathValue("id")
	p, ok := a.findProject(id)
	if !ok {
		http.Error(w, "project not found", http.StatusNotFound)
		return
	}
	loader := config.NewLoader(p.RootPath)
	rules, err := loader.LoadLogRules()
	if err != nil {
		http.Error(w, err.Error(), http.StatusInternalServerError)
		return
	}
	jsonOK(w, rules)
}

func (a *App) putProjectRules(w http.ResponseWriter, r *http.Request) {
	id := r.PathValue("id")
	p, ok := a.findProject(id)
	if !ok {
		http.Error(w, "project not found", http.StatusNotFound)
		return
	}
	var rules []model.LogRule
	if err := json.NewDecoder(r.Body).Decode(&rules); err != nil {
		http.Error(w, "invalid request", http.StatusBadRequest)
		return
	}
	loader := config.NewLoader(p.RootPath)
	if err := loader.SaveLogRules(rules); err != nil {
		http.Error(w, err.Error(), http.StatusInternalServerError)
		return
	}
	jsonOK(w, rules)
}

func (a *App) findProject(id string) (model.Project, bool) {
	a.mu.RLock()
	defer a.mu.RUnlock()
	for _, p := range a.projects {
		if p.ID == id {
			return p, true
		}
	}
	return model.Project{}, false
}

func jsonOK(w http.ResponseWriter, v any) {
	w.Header().Set("Content-Type", "application/json")
	_ = json.NewEncoder(w).Encode(v)
}

func (a *App) listServices(w http.ResponseWriter, r *http.Request) {
	a.mu.RLock()
	var services []model.Service
	for _, p := range a.projects {
		for _, svc := range p.Services {
			mgr := a.managers[p.ID]
			if mgr != nil {
				svc.Status = mgr.Status(svc.ID)
				svc.PID = mgr.PID(svc.ID)
			}
			services = append(services, svc)
		}
	}
	a.mu.RUnlock()
	jsonOK(w, services)
}

func (a *App) startService(w http.ResponseWriter, r *http.Request) {
	svcID := r.PathValue("id")
	svc, proj, ok := a.findService(svcID)
	if !ok {
		http.Error(w, "service not found", http.StatusNotFound)
		return
	}
	mgr := a.getOrCreateManager(proj.ID)
	if err := mgr.Start(svc); err != nil {
		http.Error(w, err.Error(), http.StatusInternalServerError)
		return
	}
	jsonOK(w, map[string]string{"status": "starting"})
}

func (a *App) stopService(w http.ResponseWriter, r *http.Request) {
	svcID := r.PathValue("id")
	_, proj, ok := a.findService(svcID)
	if !ok {
		http.Error(w, "service not found", http.StatusNotFound)
		return
	}
	if mgr := a.managers[proj.ID]; mgr != nil {
		mgr.Stop(svcID)
	}
	jsonOK(w, map[string]string{"status": "stopped"})
}

func (a *App) restartService(w http.ResponseWriter, r *http.Request) {
	svcID := r.PathValue("id")
	svc, proj, ok := a.findService(svcID)
	if !ok {
		http.Error(w, "service not found", http.StatusNotFound)
		return
	}
	mgr := a.getOrCreateManager(proj.ID)
	mgr.Stop(svcID)
	if err := mgr.Start(svc); err != nil {
		http.Error(w, err.Error(), http.StatusInternalServerError)
		return
	}
	jsonOK(w, map[string]string{"status": "starting"})
}

func (a *App) findService(svcID string) (model.Service, model.Project, bool) {
	a.mu.RLock()
	defer a.mu.RUnlock()
	for _, p := range a.projects {
		for _, svc := range p.Services {
			if svc.ID == svcID {
				return svc, p, true
			}
		}
	}
	return model.Service{}, model.Project{}, false
}

// startSelected 启动项目中 required 服务 + SelectedServiceIDs 中的服务，按 order 串并行。
func (a *App) startSelected(w http.ResponseWriter, r *http.Request) {
	id := r.PathValue("id")
	p, ok := a.findProject(id)
	if !ok {
		http.Error(w, "project not found", http.StatusNotFound)
		return
	}
	selectedSet := make(map[string]bool, len(p.SelectedServiceIDs))
	for _, sid := range p.SelectedServiceIDs {
		selectedSet[sid] = true
	}
	var toStart []model.Service
	for _, svc := range p.Services {
		if svc.Required || selectedSet[svc.Name] {
			toStart = append(toStart, svc)
		}
	}
	mgr := a.getOrCreateManager(p.ID)
	if err := mgr.StartGroup(toStart); err != nil {
		http.Error(w, err.Error(), http.StatusInternalServerError)
		return
	}
	jsonOK(w, map[string]string{"status": "starting"})
}

func (a *App) getOrCreateManager(projectID string) *process.Manager {
	a.mu.Lock()
	defer a.mu.Unlock()
	if mgr, ok := a.managers[projectID]; ok {
		return mgr
	}
	mgr := process.NewManager(func(e model.LogEntry) {
		a.buf.Append(e)
	})
	a.managers[projectID] = mgr
	return mgr
}

func (a *App) fetchLogs(w http.ResponseWriter, r *http.Request) {
	q := r.URL.Query()
	params := store.FetchParams{
		ServiceID: q.Get("service"),
		RunID:     q.Get("run"),
		Limit:     100,
	}
	if l := q.Get("limit"); l != "" {
		if n, err := strconv.Atoi(l); err == nil && n > 0 {
			params.Limit = n
		}
	}
	if b := q.Get("before"); b != "" {
		if n, err := strconv.ParseInt(b, 10, 64); err == nil {
			params.Before = n
		}
	}

	entries, err := a.store.Fetch(params)
	if err != nil {
		http.Error(w, err.Error(), http.StatusInternalServerError)
		return
	}
	if entries == nil {
		entries = []model.LogEntry{}
	}
	jsonOK(w, entries)
}

package api

import (
	"encoding/json"
	"net/http"

	"github.com/google/uuid"
	"github.com/gorilla/websocket"
	"github.com/superdev/agent/model"
)

var upgrader = websocket.Upgrader{
	CheckOrigin: func(r *http.Request) bool { return true },
}

func (a *App) wsLogs(w http.ResponseWriter, r *http.Request) {
	serviceID := r.URL.Query().Get("service")
	conn, err := upgrader.Upgrade(w, r, nil)
	if err != nil {
		return
	}
	defer conn.Close()

	subID := uuid.NewString()
	ch := a.buf.Subscribe(subID)
	defer a.buf.Unsubscribe(subID)

	// 先推送最近 200 条历史
	recent := a.buf.Recent(200)
	for _, e := range recent {
		if serviceID == "" || e.ServiceID == serviceID {
			data, _ := json.Marshal(e)
			if err := conn.WriteMessage(websocket.TextMessage, data); err != nil {
				return
			}
		}
	}

	for {
		select {
		case e, ok := <-ch:
			if !ok {
				return
			}
			if serviceID != "" && e.ServiceID != serviceID {
				continue
			}
			data, _ := json.Marshal(e)
			if err := conn.WriteMessage(websocket.TextMessage, data); err != nil {
				return
			}
		case <-r.Context().Done():
			return
		}
	}
}
